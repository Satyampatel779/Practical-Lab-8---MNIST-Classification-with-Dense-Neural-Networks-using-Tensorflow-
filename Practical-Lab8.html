# Define a function to perform EDA
def perform_eda(images, labels, dataset_name):
    # Display a sample of images
    fig, axes = plt.subplots(5, 5, figsize=(10,10))
    for i, ax in enumerate(axes.flat):
        ax.imshow(images[i].reshape(28, 28), cmap='gray')
        ax.set_title(f'Label: {labels[i]}')
    plt.tight_layout(pad=2.0)  # Increase padding
    plt.suptitle(f'Sample Images from {dataset_name}', fontsize=16, y=1.05)  # Adjust y position
    plt.show()

    # Display the average pixel value for each class
    fig, axes = plt.subplots(2, 5, figsize=(15,6))
    axes = axes.ravel()
    for i in np.arange(0, 10):
        axes[i].imshow(images[labels==i].mean(axis=0).reshape(28, 28), cmap='gray')
        axes[i].set_title(f'Label: {i}', fontsize=12)
    plt.subplots_adjust(wspace=0.5, hspace=0.5)  # Adjust horizontal and vertical spacing
    plt.suptitle(f'Average Image for Each Class in {dataset_name}', fontsize=16, y=1.05)  # Adjust y position
    plt.show()

# Perform EDA on the training dataset
perform_eda(train_images, train_labels, 'Training Dataset')

# Perform EDA on the testing dataset
perform_eda(test_images, test_labels, 'Testing Dataset')

What if we wanted to increase the precision for class '5', how can we do that without changing the model or retraining? provide code that exemplifies this.¶

To increase precision for class 5 without altering the model or retraining, we can change the threshold at which predictions are labeled as class 5. By raising this threshold, we make the criteria stricter, so only the most likely predictions are labeled as class 5, which can increase precision.

In [ ]:
# Defined a function to predict with a threshold
def predict_with_threshold(probabilities, threshold=0.3):
  predicted_labels = np.argmax(probabilities, axis=1)
  for i in range(len(predicted_labels)):
    if probabilities[i, 5] > threshold:  # Check probability for class '5' only
      predicted_labels[i] = 5  # Set prediction to class '5' if probability exceeds threshold
  return predicted_labels

# Calculated metrics using the threshold for class '5'
predictions_threshold = predict_with_threshold(neural_network_model.predict(normalized_test_images[:20]))
precision_threshold = precision_score(test_labels[:20], predictions_threshold, average='weighted')
print(f'\nMetrics with Threshold for Class 5:')
print(f'Precision (weighted): {precision_threshold}')

What if we wanted to increase the recall for class '5', how can we do that without changing the model or retraining? provide code that exemplifies this.¶

To increase the recall for class '5' without changing the model or retraining, we can adjust the decision threshold for this specific class 5. By default, the decision threshold is 0.5 for multi-class classification problems. If the predicted probability for a class is greater than or equal to this threshold, the instance is classified as that class.

In [ ]:
def adjust_threshold_for_recall(probabilities, class_index, new_threshold):
    # Create a new array for adjusted predictions
    adjusted_predictions_recall = np.zeros(probabilities.shape[0])
    
    # Classify instances where the probability of class '5' is greater than the new threshold as class '5'
    adjusted_predictions_recall[probabilities[:, class_index] > new_threshold] = class_index
    
    # For other instances, classify as the class with the highest probability
    adjusted_predictions_recall[probabilities[:, class_index] <= new_threshold] = np.argmax(probabilities[probabilities[:, class_index] <= new_threshold], axis=1)
    
    return adjusted_predictions_recall

# Predict the probabilities for the first 20 test images
probabilities_recall = neural_network_model.predict(normalized_test_images[:20])

# Adjust the decision threshold for class '5' to increase recall
predictions_for_recall = adjust_threshold_for_recall(probabilities_recall, 5, 0.5)

# Calculate the recall score for the adjusted predictions
recall_adjusted = recall_score(test_labels[:20], predictions_for_recall, average='weighted')

print(f'\nMetrics with Threshold for Class 5:')
